const weekdayHours = { 
  1: 8,  // Monday
  2: 8,  // Tuesday
  3: 8,  // Wednesday
  4: 8,  // Thursday
  5: 0   // Friday
};

function calculateTotalHours(startDate, endDate, holidays, weekdayHours) {
  let totalHours = 0;
  let current = new Date(startDate);
  
  while (current <= endDate) {
    const dayOfWeek = current.getDay();
    const isHoliday = holidays.includes(current.toDateString());
    
    if (dayOfWeek >= 1 && dayOfWeek <= 5 && !isHoliday) {
      totalHours += weekdayHours[dayOfWeek];
    }
    current.setDate(current.getDate() + 1);
  }
  
  return totalHours;
}

// Budget consumption
const budgetDays = totalHours / 8;
